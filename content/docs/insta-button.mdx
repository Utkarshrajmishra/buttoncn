## Installation

```npm
npx shadcn@latest add http://localhost:3000/r/pixel-button.json
```

## Usage

```tsx title="usage.tsx"
import IconButton from "../buttoncn/icon-button";

const SocialButtonPreview = () => {
  return <IconButton />;
};

export default SocialButtonPreview;
```
