## Installation

```npm
npx shadcn@latest add http://localhost:3000/components/buttocn/splash-button
```

## Usage

```tsx title="usage.tsx"
import BrokenButton from "@/components/buttoncn/broken-button";

export default function Usage() {
  return <BrokenButton />;
}
```
