## Installation

```npm
npx shadcn@latest add http://localhost/300/components/buttocn/splash-button
```


## Usage

```tsx title="usage.tsx"
import FoldButton from "@/components/buttoncn/fold-button";

export default function Usage() {
    return (
         <FoldButton >Hover Me</FoldButton>
    )
}
```
